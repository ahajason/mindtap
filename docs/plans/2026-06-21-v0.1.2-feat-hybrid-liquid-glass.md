# V0.1.2 Hybrid Liquid Glass Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在 macOS 26 真原生 per-element Liquid Glass (SwiftUI `.glassEffect()`),同时保留 React 跨平台前端 + Tauri 通用层;Win/Linux 走现有 CSS `.glass-l1/l2/l3` 兜底。

**Architecture:** 3 层 — React 19 (页面 + `useGlassOverlay` hook) / Tauri 2 Rust (commands + FFI bridge) / Swift native (SwiftUI `.glassEffect()` overlay + 整窗 `.glassBackgroundEffect()`)。Per-element 通过 Swift 在 WKWebView 上叠 `NSHostingView` overlay,位置由 React `getBoundingClientRect` + ResizeObserver 同步。FFI 用 C-ABI + JSON 字符串,不引 `swift-bridge`。

**Tech Stack:**
- Frontend: React 19 + TypeScript + Vite 7 + Tailwind 4 + vitest + @testing-library/react
- Rust: Tauri 2 + Cargo + serde + serde_json + log
- Swift: Swift 5.10 + SwiftUI + AppKit + SPM (`macOS(.v26)`)
- FFI: C-ABI + JSON 字符串 (5 个 `@_cdecl` 函数)
- 测试: XCTest (Swift), `#[test]` (Rust), vitest + jsdom (React)

## Global Constraints

> 每条来自 spec,每个 task 隐式适用。

- **macOS 最低系统**: 26.0 (Tahoe) — `.glassEffect()` / `.glassBackgroundEffect()` 要求

```swift
import XCTest
import AppKit
@testable import MindtapGlass

@MainActor
final class GlassOverlayManagerTests: XCTestCase {
    var manager: GlassOverlayManager!
    var testWindow: NSWindow!

    override func setUp() async throws {
        try await super.setUp()
        manager = GlassOverlayManager.shared
        manager.clear()  // reset state
        testWindow = NSWindow(
            contentRect: NSRect(x: 0, y: 0, width: 400, height: 400),
            styleMask: [.titled, .closable],
            backing: .buffered, defer: false
        )
    }

    override func tearDown() async throws {
        manager.clear()
        testWindow = nil
        try await super.tearDown()
    }

    func testRegisterAddsOverlay() {
        manager.attach(to: testWindow)
        manager.register(id: "a", tier: .l1, rect: CGRect(x: 10, y: 10, width: 100, height: 50))
        // 验证 overlay 添加到 contentView
        XCTAssertGreaterThanOrEqual(testWindow.contentView?.subviews.count ?? 0, 1)
    }

    func testUpdateRectChangesFrame() {
        manager.attach(to: testWindow)
        manager.register(id: "a", tier: .l1, rect: CGRect(x: 10, y: 10, width: 100, height: 50))
        manager.updateRect(id: "a", rect: CGRect(x: 20, y: 20, width: 200, height: 80))
        // 找到对应 overlay 验证 frame
        let overlay = testWindow.contentView?.subviews.first { $0.frame.origin.x == 20 }
        XCTAssertNotNil(overlay)
        XCTAssertEqual(overlay?.frame.size.width, 200)
    }

    func testUnregisterRemovesOverlay() {
        manager.attach(to: testWindow)
        manager.register(id: "a", tier: .l1, rect: .zero)
        let before = testWindow.contentView?.subviews.count ?? 0
        manager.unregister(id: "a")
        let after = testWindow.contentView?.subviews.count ?? 0
        XCTAssertEqual(after, before - 1)
    }

    func testClearRemovesAll() {
        manager.attach(to: testWindow)
        manager.register(id: "a", tier: .l1, rect: .zero)
        manager.register(id: "b", tier: .l2, rect: .zero)
        manager.register(id: "c", tier: .l3, rect: .zero)
        manager.clear()
        XCTAssertEqual(testWindow.contentView?.subviews.count ?? 0, 0)
    }

    func testRegisterWithoutAttachIsNoop() {
        // 没调 attach,register 应不崩溃也不创建 overlay
        manager.clear()
        manager.register(id: "orphan", tier: .l1, rect: .zero)
        // 没 attach,无法断言 subview 数(因为没有 window);只验证不崩溃

    func testReregisterReplacesExisting() {

        manager.register(id: "a", tier: .l1, rect: CGRect(x: 0, y: 0, width: 10, height: 10))
        let before = testWindow.contentView?.subviews.count ?? 0

        let after = testWindow.contentView?.subviews.count ?? 0
        XCTAssertEqual(after, before, "重复 register 不应增加 subview 数")
    }
```

- [ ] **Step 4: 实现 `Sources/MindtapGlass/GlassOverlayManager.swift`**

```swift
import AppKit
import SwiftUI

@MainActor
public final class GlassOverlayManager {
    public static let shared = GlassOverlayManager()
    private var overlays: [String: NSHostingView<AnyView>] = [:]
    private weak var hostWindow: NSWindow?

    private init() {}

    public func attach(to window: NSWindow) {
        self.hostWindow = window
    }

    public func register(id: String, tier: GlassTier, rect: CGRect) {
        guard let window = hostWindow else {
            print("[GlassOverlayManager] hostWindow not set; register ignored: \(id)")
            return
        }
        // 同一 id 重复 register: 先清旧
        overlays[id]?.removeFromSuperview()

        let overlay = NSHostingView(rootView: AnyView(GlassOverlay(tier: tier)))
        overlay.frame = rect
        overlay.autoresizingMask = []
        overlay.isUserInteractionEnabled = false  // 事件穿透到 WKWebView
        window.contentView?.addSubview(overlay)
        overlays[id] = overlay
    }

    public func updateRect(id: String, rect: CGRect) {
        overlays[id]?.frame = rect
    }

    public func unregister(id: String) {
        overlays[id]?.removeFromSuperview()
        overlays[id] = nil
    }

    public func clear() {
        overlays.values.forEach { $0.removeFromSuperview() }
        overlays.removeAll()
    }
}
```

```swift
import SwiftUI
import AppKit

@MainActor
public enum WindowGlassBackground {
    /// 在 WKWebView 之下插入整窗 .glassBackgroundEffect() 背景
    public static func apply(to window: NSWindow) {
        guard let contentView = window.contentView else { return }

        let host = NSHostingView(rootView: AnyView(
            Color.clear
                .glassBackgroundEffect()
                .ignoresSafeArea()
        ))
        host.frame = contentView.bounds
        host.autoresizingMask = [.width, .height]
        host.isUserInteractionEnabled = false

        contentView.addSubview(host, positioned: .below, relativeTo: nil)
    }
}
```

```ts
import { useEffect, useId } from 'react';
import type { RefObject } from 'react';
import { glassRegister, glassUpdateRect, glassUnregister, isMacOS } from './glass-invoke';
import type { GlassTier } from './glass-invoke';

/**
 * 内部 helper。给 DOM 元素挂 per-element Liquid Glass overlay。
 * - macOS 真 Tauri 环境: invoke glass_register/update_rect/unregister
 * - 其他平台 / Web 端: no-op(走 CSS 兜底)
 */
export function useGlassOverlay(
  tier: GlassTier,
  ref: RefObject<HTMLElement | null>,
  enabled = true,
): void {
  const reactId = useId();
  const id = `glass-${reactId.replace(/:/g, '')}`;

  useEffect(() => {
    if (!enabled) return;
    let cancelled = false;
    let active = false;

    (async () => {
      const mac = await isMacOS();
      if (cancelled || !mac) return;
      const el = ref.current;
      if (!el) return;

      active = true;

      const sendRect = () => {
        const r = el.getBoundingClientRect();
        glassUpdateRect(id, { x: r.x, y: r.y, w: r.width, h: r.height }).catch(() => {});
      };

      const sendRegister = () => {
        const r = el.getBoundingClientRect();
        glassRegister(id, tier, { x: r.x, y: r.y, w: r.width, h: r.height }).catch(() => {});
      };

      sendRegister();

      const ro = new ResizeObserver(sendRect);
      ro.observe(el);

      const onScroll = () => sendRect();
      window.addEventListener('scroll', onScroll, { passive: true });

      // 暴露 cleanup
      cleanup = () => {
        ro.disconnect();
        window.removeEventListener('scroll', onScroll);
        if (active) glassUnregister(id).catch(() => {});
        active = false;
      };
    })();

    let cleanup: (() => void) | undefined;
    return () => {
      cancelled = true;
      cleanup?.();
    };
  }, [tier, enabled, ref, id]);
}
```

```tsx
import { forwardRef, useRef, type HTMLAttributes } from 'react';
import { cva, type VariantProps } from 'class-variance-authority';
import { cn } from '@/lib/utils';
import { useGlassOverlay } from '@/lib/use-glass-overlay';

const cardVariants = cva(
  'rounded-[var(--radius-card)] transition-all duration-base ease-[var(--ease-out)]',
  {
    variants: {
      tier: {
        l1: 'glass-l1',
        l2: 'glass-l2',
        l3: 'glass-l3',
      },
    },
    defaultVariants: {
      tier: 'l1',
    },
  }
);

type CardTier = NonNullable<VariantProps<typeof cardVariants>['tier']>;

export interface CardProps extends HTMLAttributes<HTMLDivElement> {
  tier?: CardTier;
}

export const Card = forwardRef<HTMLDivElement, CardProps>(
  ({ className, tier, ...props }, ref) => {
    const innerRef = useRef<HTMLDivElement | null>(null);
    useGlassOverlay(tier ?? 'l1', innerRef);

    return (
      <div
        ref={(el) => {
          innerRef.current = el;
          if (typeof ref === 'function') ref(el);
          else if (ref) ref.current = el;
        }}
        className={cn(cardVariants({ tier }), className)}
        {...props}
      />
    );
  }
);
Card.displayName = 'Card';
```

```tsx
export const Button = forwardRef<HTMLButtonElement, ButtonProps>(
  ({ className, variant = 'primary', size = 'md', ...props }, ref) => {
    const innerRef = useRef<HTMLButtonElement | null>(null);
    // secondary 和 icon variants 挂 overlay (l1);primary (实色) 和 ghost (透明) 不挂
    const glassTier = variant === 'primary' || variant === 'ghost' ? null : 'l1';

          if (typeof ref === 'function') ref(el);
          else if (ref) ref.current = el;
        }}
        className={cn(buttonVariants({ variant, size }), className)}
        {...props}
      />
    );
  }
```
