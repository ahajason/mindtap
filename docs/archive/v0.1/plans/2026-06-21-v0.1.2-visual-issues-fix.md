> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Fix all V0.1.2 visual issues across 5 priority tiers (P0 foundation / P1 active-inactive / P1 content-layer / P1 geometry / P1 spacing-typography) by implementing the design specs in `docs/research/2026-06-21-v0.1.2-glass-issues/1-design/` and the solutions in `docs/research/2026-06-21-v0.1.2-glass-issues/2-issues/`.

Spacing migrated to `var(--spacing-N)` + Tailwind config mapping for zero-touch token propagation.

**Tech Stack:** Tauri 2 + React 19 + TypeScript + Vite 7 + Tailwind 4 + shadcn (badge/button/card/dialog/input/label/separator/sonner/tabs/textarea/tooltip). Vitest + Testing Library.

```ts
// src/hooks/useWindowActive.ts
import { useEffect, useState } from 'react';

export function useWindowActive(): boolean {
  const [active, setActive] = useState(() =>
    typeof document !== 'undefined' ? document.hasFocus() : true
  );

  useEffect(() => {
    const onFocus = () => {
      setActive(true);
      document.documentElement.dataset.windowActive = 'true';
    };
    const onBlur = () => {
      setActive(false);
      document.documentElement.dataset.windowActive = 'false';
    };
    const onVisibility = () => {
      const next = !document.hidden && document.hasFocus();
      setActive(next);
      document.documentElement.dataset.windowActive = String(next);
    };

    window.addEventListener('focus', onFocus);
    window.addEventListener('blur', onBlur);
    document.addEventListener('visibilitychange', onVisibility);

    // 初始同步
    onVisibility();

    return () => {
      window.removeEventListener('focus', onFocus);
      window.removeEventListener('blur', onBlur);
      document.removeEventListener('visibilitychange', onVisibility);
    };
  }, []);

  return active;
}
```

```tsx
// src/components/layout/NavLink.tsx
import { NavLink as RouterNavLink, useLocation } from 'react-router-dom';
import { cn } from '@/lib/utils';
import type { ReactNode } from 'react';

interface SidebarNavLinkProps {
  to: string;
  icon: ReactNode;
  label: string;
}

export function SidebarNavLink({ to, icon, label }: SidebarNavLinkProps) {
  const { pathname } = useLocation();
  const isActive = pathname === to || pathname.startsWith(to + '/');

  return (
    <RouterNavLink
      to={to}
      aria-current={isActive ? 'page' : undefined}
      className={cn(
        'flex items-center gap-2 px-3 py-2 rounded-md text-sm',
        'transition-colors duration-150 ease-[var(--ease-out)]',
        'focus-visible:outline-none focus-visible:ring-2',
        'focus-visible:ring-primary focus-visible:ring-offset-2',
        isActive
          ? 'bg-primary text-white font-semibold border-l-2 border-primary'
          : 'text-text-2 hover:bg-black/[0.04] hover:text-text-1'
      )}
    >
      {icon}
      <span>{label}</span>
    </RouterNavLink>
  );
}
```

```tsx
import { forwardRef, type HTMLAttributes } from 'react';
import { cn } from '@/lib/utils';

type CardPadding = 'none' | 'sm' | 'md' | 'lg';

interface CardProps extends HTMLAttributes<HTMLDivElement> {
  tier?: 'l1' | 'l2' | 'l3';
  padding?: CardPadding;
}

const paddingMap: Record<CardPadding, string> = {
  none: '',
  sm: 'p-[var(--spacing-3)]',
  md: 'p-[var(--spacing-4)]',
  lg: 'p-[var(--spacing-6)]',
};

export const Card = forwardRef<HTMLDivElement, CardProps>(
  ({ className, tier, padding = 'md', ...props }, ref) => (
    <div
      ref={ref}
      className={cn(
        tier && `glass-${tier}`,
        'rounded-[var(--radius-card)]',
        paddingMap[padding],
        className
      )}
      {...props}
    />
  )
);
Card.displayName = 'Card';
```
