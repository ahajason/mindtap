D4 3 种 record 关联设计 · 你选

方案 A · 触发器自动同步（推荐）

record 表 + SQLite 触发器自动写入/更新，业务层不需双写。
UNIQUE(kind, source_id) 防止重复。

方案 B · 业务层双写

业务层每次创建 task/idea/check_in 同步写 record。
简单直接，但易遗漏。

方案 C · 纯视图 JOIN

不建 record 表，用 UNION ALL VIEW 聚合 3 源表。
无冗余但查询性能受限。

📌 方案 A · 触发器自动同步（推荐）完整 Schema

-- ===========================================
-- 3 源表（V15 已有）
-- ===========================================
CREATE TABLE task (
  id                  INTEGER PRIMARY KEY AUTOINCREMENT,
  content             TEXT    NOT NULL,
  status              TEXT    NOT NULL DEFAULT 'pending',
  duration_ms         INTEGER NOT NULL DEFAULT 0,
  first_focused_at    INTEGER,
  focus_started_at    INTEGER,
  paused_at           INTEGER,
  focused_count       INTEGER NOT NULL DEFAULT 0,
  due_at              INTEGER,
  created_at          INTEGER NOT NULL,
  updated_at          INTEGER NOT NULL,
  archived_at         INTEGER
);

CREATE TABLE idea (
  id          INTEGER PRIMARY KEY AUTOINCREMENT,
  content     TEXT    NOT NULL,
  created_at  INTEGER NOT NULL
);

CREATE TABLE check_in (
  id          INTEGER PRIMARY KEY AUTOINCREMENT,
  habit       TEXT    NOT NULL,
  note        TEXT,
  checked_at  INTEGER NOT NULL
);

-- ===========================================
-- 通用 record 表（统一列表）
-- ===========================================
CREATE TABLE record (
  id          INTEGER PRIMARY KEY AUTOINCREMENT,
  kind        TEXT    NOT NULL,         -- 'task' | 'idea' | 'check_in'
  source_id   INTEGER NOT NULL,         -- 指向 task/idea/check_in.id
  content     TEXT    NOT NULL,         -- 实时同步源表 content（task.content / idea.content / check_in.habit+note）
  status      TEXT,                     -- task 有 status；idea/check_in 为 NULL
  created_at  INTEGER NOT NULL,         -- 复用源表时间戳
  UNIQUE(kind, source_id)               -- 同源只能一条
);
CREATE INDEX idx_record_created_at ON record(created_at);
CREATE INDEX idx_record_kind ON record(kind);
CREATE INDEX idx_record_status ON record(status);

-- ===========================================
-- 触发器：INSERT 时自动写 record
-- ===========================================
CREATE TRIGGER task_after_insert AFTER INSERT ON task
BEGIN
  INSERT INTO record (kind, source_id, content, status, created_at)
  VALUES ('task', NEW.id, NEW.content, NEW.status, NEW.created_at);
END;

CREATE TRIGGER idea_after_insert AFTER INSERT ON idea
BEGIN
  INSERT INTO record (kind, source_id, content, status, created_at)
  VALUES ('idea', NEW.id, NEW.content, NULL, NEW.created_at);
END;

CREATE TRIGGER check_in_after_insert AFTER INSERT ON check_in
BEGIN
  INSERT INTO record (kind, source_id, content, status, created_at)
  VALUES ('check_in', NEW.id,
          NEW.habit || COALESCE(' · ' || NEW.note, ''),
          NULL, NEW.checked_at);
END;

-- ===========================================
-- 触发器：UPDATE 时同步 record.content/status
-- ===========================================
CREATE TRIGGER task_after_update AFTER UPDATE OF content, status ON task
BEGIN
  UPDATE record
  SET content = NEW.content, status = NEW.status
  WHERE kind = 'task' AND source_id = NEW.id;
END;

-- idea 只改 content
CREATE TRIGGER idea_after_update AFTER UPDATE OF content ON idea
BEGIN
  UPDATE record
  SET content = NEW.content
  WHERE kind = 'idea' AND source_id = NEW.id;
END;

-- ===========================================
-- 触发器：DELETE 时级联删除 record
-- ===========================================
CREATE TRIGGER task_after_delete AFTER DELETE ON task
BEGIN
  DELETE FROM record WHERE kind = 'task' AND source_id = OLD.id;
END;

CREATE TRIGGER idea_after_delete AFTER DELETE ON idea
BEGIN
  DELETE FROM record WHERE kind = 'idea' AND source_id = OLD.id;
END;

CREATE TRIGGER check_in_after_delete AFTER DELETE ON check_in
BEGIN
  DELETE FROM record WHERE kind = 'check_in' AND source_id = OLD.id;
END;

主窗列表 SQL（按 record 表单表查询）

-- 全部记录（最新优先）
SELECT id, kind, content, status, created_at
FROM record
ORDER BY created_at DESC
LIMIT 50;

-- 仅任务
SELECT * FROM record WHERE kind = 'task' ORDER BY created_at DESC;

-- 今日
SELECT * FROM record
WHERE created_at >= strftime('%s', 'now', 'start of day') * 1000
ORDER BY created_at DESC;

-- 任务按状态分组
SELECT status, COUNT(*) FROM record
WHERE kind = 'task' AND created_at >= ...
GROUP BY status;

关键优势：单表查询，无需 UNION 3 源，索引高效，点击列表项 → 用 kind + source_id 跳转源表详情。

⚖ 3 方案对比

维度 A 触发器自动同步（推荐） B 业务层双写 C 纯视图 UNION
业务层负担 ✅ 零负担（触发器自动） ⚠️ 每次创建/更新要双写 ✅ 零负担
列表查询性能 ✅ 单表 + 索引 ✅ 单表 + 索引 ⚠️ UNION 3 表 + 排序
数据一致性 ✅ 触发器保证 ⚠️ 双写遗漏风险 ✅ 视图自动
源表删除/归档 ⚠️ 需级联触发器 ⚠️ 业务层处理 ✅ 视图自动过滤
调试复杂度 ⚠️ 触发器略复杂 ✅ 简单直接 ✅ 视图简单
扩展性 ✅ 加表只加触发器 ⚠️ 加表要改业务层 ⚠️ 改视图 SQL
迁移 V1.2 数据 ✅ 一次性脚本回填 record ✅ 一次性脚本回填 ✅ 视图无数据

推荐 A：单表查询快 + 业务层零负担 + 长期扩展容易。触发器复杂度可接受（10 个 SQL 行）。
